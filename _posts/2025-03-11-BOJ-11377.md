```c++
#include <bits/stdc++.h>
#define fastio cin.tie(0)->ios::sync_with_stdio(0)

using namespace std;
using ll = long long;
using pi = pair<int, int>;
using pll = pair<ll, ll>;
using vi = vector<int>;
using vvi = vector<vector<int>>;
using vvll = vector<vector<ll>>;
using vll = vector<ll>;
using vpi = vector<pi>;

constexpr int MAX = 1'000;

int N, M, K, match[MAX + 1], is_visited[MAX + 1];
vi graph[MAX + 1];

bool dfs(int node) {
    for (auto next : graph[node]) {
        if (is_visited[next]) continue;
        is_visited[next] = true;
        if (match[next] == node) continue;
        if (match[next] == 0 || dfs(match[next])) {
            match[next] = node;
            return true;
        }
    }
    return false;
}

int main() {
    fastio;

    cin >> N >> M >> K;
    for (int i = 1; i <= N; ++i) {
        int n;
        cin >> n;
        while (n--) {
            int m;
            cin >> m;
            graph[i].push_back(m);
        }
    }

    for (int i = 1; i <= N; ++i) {
        memset(is_visited, 0, sizeof(is_visited));
        dfs(i);
    }

    for (int i = 1; i <= N; ++i) {
        if (!K) break;
        memset(is_visited, 0, sizeof(is_visited));
        if (dfs(i)) K--;
    }

    int ans = 0;
    for (int i = 1; i <= M; ++i) {
        ans += !!match[i];
    }
    cout << ans;
}
```
