#include <bits/stdc++.h>
#define fastio cin.tie(0)->ios::sync_with_stdio(0)

using namespace std;
using ll = long long;
using pi = pair<int, int>;
using pll = pair<ll, ll>;
using vi = vector<int>;
using vvi = vector<vector<int>>;
using vvll = vector<vector<ll>>;
using vll = vector<ll>;
using vpi = vector<pi>;

constexpr int MAX = 50'000;

int N, K, p[MAX + 1];
vi graph[MAX + 1];

void bfs(int node) {
    queue<int> q;
    q.push(node);
    p[node] = 1;
    while (!q.empty()) {
        int curr = q.front();
        q.pop();
        for (auto next : graph[curr]) {
            if (p[next]) continue;
            q.push(next);
            p[next] = curr;
        }
    }
}

int main() {
    fastio;

    cin >> N >> K;
    for (int i = 0; i < N; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
    }
    bfs(1);
    if (p[K]) {
        vi ans;
        while (true) {
            ans.push_back(K);
            if (p[K] == K) break;
            K = p[K];
        }
        cout << ans.size() << "\n";
        for (int i = ans.size() - 1; i >= 0; --i) {
            cout << ans[i] << "\n";
        }
    } else {
        cout << -1;
    }
}
