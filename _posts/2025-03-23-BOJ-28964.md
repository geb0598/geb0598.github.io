    bool is_possible = true;

        if (b > e) {
            is_possible = false;
            break;
        }
        l = max(l + 1, b);
        r = max(r, e);
        if (l > r) {
            is_possible = false;
            break;
        }
    }
    cout << (is_possible ? "YES" : "NO");
}
